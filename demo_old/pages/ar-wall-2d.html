Model will be placed on the detected wall.

import { WallARSession } from "@egjs/view3d";

const wallSession = new WallARSession({
  // Root element for dom-overlay above AR session
  overlayRoot: overlay,
  // Element to show loading indicator until the floor is found.
  loadingEl: "#loading",
  // Prevent entering to AR session without dom-overlay feature.
  forceOverlay: true,
}).on("start", () => {
  overlay.style.display = "flex";
}).on("end", () => {
  overlay.style.display = "none";
});
view3d.xr.addSession(wallSession);

arButton.addEventListener("click", async () => {
  view3d.xr.enter().catch(err => {
    alert(err);
    console.error(err);
  });
});
