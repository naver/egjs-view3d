## Example
### autoplay: true
<View3D src="/model/bunny.drc" autoplay={true} preset={new WhiteLightPreset()} />

```ts
import View3D, { WhiteLightPreset } from "@egjs/view3d";

new View3D("#view3d-canvas", {
  src: "/model/bunny.drc",
  preset: new WhiteLightPreset(),
  autoplay: true
})
```
